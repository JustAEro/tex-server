\newcommand{\reportheader}[5]{
	\fancyhead{
	\begin{center}
	\begin{tabularx}{\textwidth}{XX}
	\Large\textbf{#1}&\multicolumn{1}{l}{\hfill\includegraphics[height=1cm]{/home/node/assets/icon.png}} \\
	\multicolumn{2}{l}{\LARGE\textbf{#2}}\\
	\normalsize Surname, Name & #3\\
	Department & #4 \\
	Report & #5\\
	\end{tabularx}
	\hrule
	\end{center}
	}
}

% footer command, for single line footer
\newcommand{\singlelinefooter}[4]{
    \fancyfoot{
    \begin{tabularx}{\textwidth}[b]{|RR|RR|}
    \hline
    \includegraphics[height=1cm,width=3cm,keepaspectratio]{#1} & #2 &  \includegraphics[height=1cm,width=3cm,keepaspectratio]{#3} & #4\\
    \hline
    \multicolumn{4}{r}{Page \thepage}
    \end{tabularx}
    }
}
